Explanation

In this assigment, annual data of S&P 500 is imported from WRDS with variables of Earning Per Share, EBIT (Earnings Before Interest and Taxes), Closing Prices, Market Cap, etc. All missing values are removed and the time horizon is from September 2014 to September 2018, in total of 4 years time period. Summaries of data are also provided as well as the plot.

Method

In order to do 3-year average of the lagged change in percent earnings, we take 4-year time period so that we can compute changes between each year and get 3 changes. For change in percent earnings, if a company has less than 4 years' records, we simply do a 2-year average; if a company has only 1 year's record, we set the average percentage change equal to 0 for that company instead of removing the company.
The E/P ratio is computed by dividing Earning Per Share by Closing Price for each company in each year.
Log Market Cap is also computed as the average log Market Cap for each company in their existing years during the 4-year time horizon. In the plot, each dot represents a company with an assigned color and the size of each dot represents its log Market Cap size. The y axis is the E/P ratio and the x axis is the Lagged Change in Percent Earnings.

Statistical Conclusion

Based on the plot, we can tell that most of the firms are clustering around 0 - 0.1 change of percent earnings and 0.03 - 0.6 E/P ratio. In terms of Market Cap, we cannot really tell how the size of market cap can influence a firm's change in percent earnings and E/P ratio within S&P 500 companies. Thus, we conlude that the Market Cap of a company in S&P 500 may not cause much effect on a firm's profitability based on the 10th and 90th quantile of the data from S&P 500.

Computer Code 

library("readxl")
snp = read_excel("PE vs MV.xlsx")
snp = snp[complete.cases(snp),]

#EP calculating --------------------------------------------------
com = length(unique(snp$`Ticker Symbol`))
n = length(snp$`Data Year - Fiscal`)
temp = c()
EP = c() #matrix(0,nrow = 1, ncol = com)
for (i in 1:(n-1))
{
  if (snp$`Company Name`[i] == snp$`Company Name`[i+1]){
    temp = append(temp,snp$`Earnings Per Share (Basic) - Excluding Extraordinary Items`[i]/snp$`Price Close - Annual - Fiscal`[i])
  } else if (snp$`Company Name`[i] != snp$`Company Name`[i+1]){
    temp = append(temp,snp$`Earnings Per Share (Basic) - Excluding Extraordinary Items`[i]/snp$`Price Close - Annual - Fiscal`[i])
    EP = c(EP,mean(temp))
    temp = c()
  }
}
EP = c(EP,mean(append(temp,snp$`Earnings Per Share (Basic) - Excluding Extraordinary Items`[i+1]/snp$`Price Close - Annual - Fiscal`[i+1])))
#make change if the last 2 observations belongs to 2 different companies

#percentage change calculating ------------------------------------
temp = c()
delta = c()
j = 1
for (i in 1:(n-1))
{
  if (snp$`Company Name`[i] == snp$`Company Name`[i+1]){
    temp = append(temp,snp$`Earnings Before Interest and Taxes`[i+1]/snp$`Earnings Before Interest and Taxes`[i] - 1)
  } else if (snp$`Company Name`[i] != snp$`Company Name`[i+1]){
    temp = append(temp,snp$`Earnings Before Interest and Taxes`[i]/snp$`Earnings Before Interest and Taxes`[i] - 1)
    delta = c(delta,mean(temp))
    temp = c()
    j = j + 1
  }
}
delta = c(delta,mean(temp))

#market cap avg calculating ------------------------------------
temp = c()
mcap = c()
j = 1
for (i in 1:(n-1))
{
  if (snp$`Company Name`[i] == snp$`Company Name`[i+1]){
    temp = append(temp,log(snp$`Market Value - Total - Fiscal`[i]))
  } else if (snp$`Company Name`[i] != snp$`Company Name`[i+1]){
    temp = append(temp,log(snp$`Market Value - Total - Fiscal`[i]))
    mcap = c(mcap,mean(temp))
    temp = c()
    j = j + 1
  }
}
mcap = c(mcap,mean(append(temp,log(snp$`Market Value - Total - Fiscal`[i+1]))))

#combine---------------------------------------------------------
All = data.frame(EP,delta,mcap)
interval1 = quantile(All$delta, probs = seq(0,1,0.01))
ready = All[which(All$delta >= interval1[11]),]
ready = ready[which(ready$delta <= interval1[length(interval1)-10]),]
interval2 = quantile(ready$EP, probs = seq(0,1,0.01))
ready = ready[which(ready$EP >= interval2[11]),]
ready = ready[which(ready$EP <= interval2[length(interval1)-10]),]
names(ready) = c("EP Ratio","Lagged Change in Percent Earnings",
                 "Market Cap")

#ggplot ---------------------------------------------------------
library("ggplot2")
Companies = c(1:length(ready$EP))
ggplot(ready, aes(x = `Lagged Change in Percent Earnings`,
                  y = `EP Ratio`)) + 
  geom_point(aes(size = `Market Cap`, color = Companies))

summary(ready)
